# –û–±—Ä–∞–±–æ—Ç–∫–∞ –∫–ª–∏–∫–æ–≤ –ø–æ –∫–Ω–æ–ø–∫–∞–º
@bot.callback_query_handler(func=lambda call: True)
def handle_button_click(call):
    chat_id = call.message.chat.id
    
    if call.data == 'quote':
        quote = random.choice(QUOTES)
        bot.send_message(chat_id, quote)
        
    elif call.data == 'advice':
        advice = random.choice(ADVICES)
        bot.send_message(chat_id, advice)
        
    elif call.data == 'fact':
        fact = random.choice(FACTS)
        bot.send_message(chat_id, fact)
        
    elif call.data == 'quiz':
        question = random.choice(list(QUESTIONS.keys()))  # –í—ã–±–æ—Ä –≤–æ–ø—Ä–æ—Å–∞
        answers = QUESTIONS[question]  # –í–∞—Ä–∏–∞–Ω—Ç—ã –æ—Ç–≤–µ—Ç–æ–≤
        correct_answer = ANSWERS[question]  # –ü—Ä–∞–≤–∏–ª—å–Ω—ã–π –æ—Ç–≤–µ—Ç
        random.shuffle(answers)  # –ü–µ—Ä–µ–º–µ—à–∏–≤–∞–Ω–∏–µ –ø–æ—Ä—è–¥–∫–∞ –æ—Ç–≤–µ—Ç–æ–≤
        
        # –§–æ—Ä–º–∏—Ä–æ–≤–∞–Ω–∏–µ inline-–∫–Ω–æ–ø–æ–∫
        keyboard = types.InlineKeyboardMarkup()
        for ans in answers:
            btn = types.InlineKeyboardButton(ans, callback_data=f"answer_{ans}_{correct_answer}")
            keyboard.add(btn)
        
        bot.send_message(chat_id, question, reply_markup=keyboard)
